/** * ═══════════════════════════════════════════════════════════════════ * MISAGI S.A.C. — Apps Script Backend v3 * Programación de Vehículos · Base de Datos Google Sheets * ═══════════════════════════════════════════════════════════════════ * * INSTRUCCIONES: * 1. Extensiones → Apps Script → pega este código → Guardar (Ctrl+S) * 2. Ejecuta setupSheets() UNA sola vez → crea las 4 hojas con * encabezados y 100 filas vacías en cada una * 3. Llena las hojas con tus datos reales * 4. Implementar → Nueva implementación → App web * - Ejecutar como: Yo * - Acceso: Cualquier usuario (Anyone) * 5. Copia la URL → pégala en CFG.APPS_SCRIPT_URL del HTML */ // ───────────────────────────────────────────── // GET // ───────────────────────────────────────────── function doGet(e) { const params = (e && e.parameter) ? e.parameter : {}; const action = params.action || 'getData'; const callback = params.callback || ''; // JSONP callback let result; try { switch (action) { case 'getData': result = getAllData(); break; case 'conductores': result = getConductores(); break; case 'tractos': result = getTractos(); break; case 'tolvas': result = getTolvas(); break; case 'rutas': result = getRutas(); break; default: result = { error: 'Accion no reconocida: ' + action }; } } catch (err) { result = { error: err.message }; } const json = JSON.stringify(result); // Si viene parametro callback → responder como JSONP (evita CORS) if (callback) { return ContentService .createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } // Sin callback → JSON normal return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } // ───────────────────────────────────────────── // POST // ───────────────────────────────────────────── function doPost(e) { try { const payload = JSON.parse(e.postData.contents); const action = payload.action || ''; let result; if (action === 'addConductor') result = addRow('CONDUCTORES', payload.data); else if (action === 'addTracto') result = addRow('TRACTOS', payload.data); else if (action === 'addTolva') result = addRow('TOLVAS', payload.data); else result = { error: 'Acción POST no reconocida' }; return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ error: err.message })).setMimeType(ContentService.MimeType.JSON); } } // ───────────────────────────────────────────── // GET ALL DATA // ───────────────────────────────────────────── function getAllData() { return { conductores: getConductores(), tractos: getTractos(), tolvas: getTolvas(), rutas: getRutas(), timestamp: new Date().toISOString() }; } // ───────────────────────────────────────────── // CONDUCTORES // Col: N | NOMBRE(SELECTOR) | APELLIDOS | NOMBRES | LICENCIA | DNI | EMPRESA // ───────────────────────────────────────────── function getConductores() { const sheet = getSheet('CONDUCTORES'); const last = sheet.getLastRow(); if (last < 2) return []; const data = sheet.getRange(2, 1, last - 1, 6).getValues(); const result = []; for (let i = 0; i < data.length; i++) { const r = data[i]; // Estructura exacta: N° | APELLIDOS | NOMBRES | LICENCIA | DNI | EMPRESA const apellidos = String(r[1] || '').trim().toUpperCase(); const nombres = String(r[2] || '').trim(); if (!apellidos && !nombres) continue; // fila vacía result.push({ n: r[0] || (i + 1), apellidos: apellidos, nombres: nombres, licencia: String(r[3] || '').trim().toUpperCase(), dni: String(r[4] || '').trim(), empresa: String(r[5] || 'MISAGI').trim().toUpperCase() }); } return result; } function getTractos() { const sheet = getSheet('TRACTOS'); const last = sheet.getLastRow(); if (last < 2) return []; const data = sheet.getRange(2, 1, last - 1, 6).getValues(); const result = []; for (let i = 0; i < data.length; i++) { const r = data[i]; if (!String(r[1]).trim()) continue; // sin placa = vacío result.push({ n: r[0] || (i + 1), placa: String(r[1] || '').trim().toUpperCase(), marca: String(r[2] || '').trim().toUpperCase(), anio: String(r[3] || '').trim(), mtc: String(r[4] || '').trim().toUpperCase(), empresa: String(r[5] || 'MISAGI').trim().toUpperCase() }); } return result; } // ───────────────────────────────────────────── // TOLVAS // Col: N | PLACA | AÑO MODELO // ───────────────────────────────────────────── function getTolvas() { const sheet = getSheet('TOLVAS'); const last = sheet.getLastRow(); if (last < 2) return []; const data = sheet.getRange(2, 1, last - 1, 3).getValues(); const result = []; for (let i = 0; i < data.length; i++) { const r = data[i]; if (!String(r[1]).trim()) continue; // sin placa = vacío result.push({ n: r[0] || (i + 1), placa: String(r[1] || '').trim().toUpperCase(), anio: String(r[2] || '').trim() }); } return result; } // ───────────────────────────────────────────── // RUTAS // Col: N | NOMBRE RUTA | ICONO | KM // ───────────────────────────────────────────── function getRutas() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('RUTAS'); if (!sheet) return defaultRutas(); const last = sheet.getLastRow(); if (last < 2) return defaultRutas(); const data = sheet.getRange(2, 1, last - 1, 4).getValues(); const result = []; for (let i = 0; i < data.length; i++) { const r = data[i]; if (!String(r[1]).trim()) continue; // sin nombre = vacío result.push({ n: r[0] || (i + 1), nombre: String(r[1] || '').trim(), icono: String(r[2] || '🛣️').trim(), km: String(r[3] || '').trim() }); } return result.length ? result : defaultRutas(); } function defaultRutas() { return [ { nombre: 'AQP → LIMA → AQP', icono: '🛣️', km: '1,030 km' }, { nombre: 'AQP → ILO', icono: '⚓', km: '230 km' }, { nombre: 'AQP → MATARANI', icono: '🏭', km: '140 km' }, { nombre: 'AQP → CUSCO', icono: '🗻', km: '520 km' }, { nombre: 'AQP → PUNO', icono: '🌊', km: '295 km' }, { nombre: 'LOCAL / MINERA', icono: '⛏️', km: 'Variable' } ]; } // ───────────────────────────────────────────── // HELPERS // ───────────────────────────────────────────── function getSheet(name) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(name); return sheet || createSheet(name); } function addRow(sheetName, rowData) { const sheet = getSheet(sheetName); // Busca la primera fila verdaderamente vacía (col 2 vacía) const last = sheet.getLastRow(); const data = sheet.getRange(2, 2, Math.max(last - 1, 1), 1).getValues(); let targetRow = last + 1; for (let i = 0; i < data.length; i++) { if (!String(data[i][0]).trim()) { targetRow = i + 2; break; } } sheet.getRange(targetRow, 1, 1, rowData.length).setValues([rowData]); return { success: true, row: targetRow }; } // ───────────────────────────────────────────── // SETUP ← EJECUTAR UNA SOLA VEZ // Crea las 4 hojas con encabezados + 100 filas vacías // ───────────────────────────────────────────── function setupSheets() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const SHEETS = { CONDUCTORES: { cols: ['N°', 'APELLIDOS', 'NOMBRES', 'LICENCIA', 'DNI', 'EMPRESA'], widths: [40, 220, 180, 120, 100, 100] }, TRACTOS: { cols: ['N°', 'PLACA', 'MARCA', 'AÑO MODELO', 'MTC', 'EMPRESA'], widths: [40, 100, 180, 100, 140, 100] }, TOLVAS: { cols: ['N°', 'PLACA', 'AÑO MODELO'], widths: [40, 100, 100] }, RUTAS: { cols: ['N°', 'NOMBRE RUTA', 'ICONO (emoji)', 'DISTANCIA KM'], widths: [40, 220, 100, 110] } }; Object.keys(SHEETS).forEach(name => { let sheet = ss.getSheetByName(name); // Crear si no existe if (!sheet) { sheet = ss.insertSheet(name); Logger.log('✅ Creada: ' + name); } else { Logger.log('ℹ️ Ya existe: ' + name + ' — solo se ajustan encabezados y filas'); } const cfg = SHEETS[name]; const nCol = cfg.cols.length; // ── Encabezado fila 1 ── const hdrRange = sheet.getRange(1, 1, 1, nCol); hdrRange.setValues([cfg.cols]); hdrRange.setBackground('#1e4d8c'); hdrRange.setFontColor('#ffffff'); hdrRange.setFontWeight('bold'); hdrRange.setFontFamily('Arial'); hdrRange.setFontSize(10); hdrRange.setHorizontalAlignment('center'); hdrRange.setVerticalAlignment('middle'); sheet.setRowHeight(1, 28); // ── 100 filas vacías (filas 2 a 101) ── const ROWS = 100; const emptyData = []; for (let r = 0; r < ROWS; r++) { emptyData.push(new Array(nCol).fill('')); } // Solo escribe si el contenido no tiene datos reales (protege info existente) const existingLast = sheet.getLastRow(); if (existingLast < 2) { sheet.getRange(2, 1, ROWS, nCol).setValues(emptyData); } else { // Extender hasta fila 101 si hay menos de 100 filas de datos const needed = 101 - existingLast; if (needed > 0) { sheet.getRange(existingLast + 1, 1, needed, nCol) .setValues(new Array(needed).fill(new Array(nCol).fill(''))); } } // ── Ancho de columnas ── cfg.widths.forEach((w, idx) => sheet.setColumnWidth(idx + 1, w)); // ── Formato de filas de datos (2-101) ── const dataRange = sheet.getRange(2, 1, ROWS, nCol); dataRange.setFontFamily('Arial'); dataRange.setFontSize(10); dataRange.setVerticalAlignment('middle'); sheet.setRowHeights(2, ROWS, 22); // ── Alternar color de fila para lectura fácil ── for (let r = 2; r <= 101; r++) { sheet.getRange(r, 1, 1, nCol).setBackground(r % 2 === 0 ? '#f5f9ff' : '#ffffff'); } // ── Congelar encabezado ── sheet.setFrozenRows(1); // ── Borde completo ── sheet.getRange(1, 1, 101, nCol) .setBorder(true, true, true, true, true, true, '#cccccc', SpreadsheetApp.BorderStyle.SOLID); }); // Orden de hojas: poner las 4 al frente const orden = ['CONDUCTORES', 'TRACTOS', 'TOLVAS', 'RUTAS']; orden.reverse().forEach(name => { const s = ss.getSheetByName(name); if (s) ss.setActiveSheet(s), ss.moveActiveSheet(1); }); Logger.log(''); Logger.log('🚛 MISAGI — Setup completo'); Logger.log('4 hojas listas con encabezados y 100 filas vacías cada una.'); Logger.log('Ahora llena los datos directamente en el Spreadsheet.'); } // ───────────────────────────────────────────── // TEST — para verificar antes de implementar // Ejecuta desde el editor → revisa los Logs // ───────────────────────────────────────────── function testDoGet() { const fakeEvent = { parameter: { action: 'getData' } }; const result = doGet(fakeEvent); const parsed = JSON.parse(result.getContent()); Logger.log('CONDUCTORES: ' + parsed.conductores.length); Logger.log('TRACTOS: ' + parsed.tractos.length); Logger.log('TOLVAS: ' + parsed.tolvas.length); Logger.log('RUTAS: ' + parsed.rutas.length); Logger.log('Timestamp: ' + parsed.timestamp); Logger.log('--- Primer conductor: ' + JSON.stringify(parsed.conductores[0])); Logger.log('--- Primer tracto: ' + JSON.stringify(parsed.tractos[0])); }
